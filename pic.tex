		output/.style   = {coordinate} % Output
	}

	node [block, below of=times1,xshift=1.5cm] (delay1) {延迟}
	node [block, below of=times2,xshift=1.5cm] (delay2) {Delay T}
	node [output, right of=times2,name=output1, xshift=2.5cm]{}

	\draw [->] (times1) -- (times2);
